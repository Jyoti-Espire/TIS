CREATE TABLE Questions ( 
Total_no_of_questions int,
    HRQ varchar(255)  NOT NULL,
    ITQ varchar(255) NOT NULL,
    ACQ varchar(255) NOT NULL,
);


CREATE TABLE HRQ (
    Total_no_of_questions int,
    meet_deadline varchar(30)  NOT NULL,
    Clients varchar(30) NOT NULL,
    Recruited_in_dept varchar(30) NOT NULL,
	Skills varchar(30) NOT NULL,
	Behaviour varchar(30) NOT NULL,
	Time_management varchar(30) NOT NULL,
    
);
create PROCEDURE Proc_HRQ_Insert
AS
BEGIN Transaction
   declare  @H_Total_no_of_questions Int
   declare @H_meet_deadline varchar(30)
   declare @H_Clients varchar(30)
   declare @H_Recruited_in_dept varchar(30)
   declare @H_Skills varchar(30)
   declare @H_Behaviour varchar(30)
   declare @H_Time_management varchar(30)




   insert into HRQ(Total_no_of_questions,meet_deadline,Clients,Recruited_in_dept,Skills,Behaviour,Time_management) values(@H_Total_no_of_questions,@H_meet_deadline,@H_Clients,@H_Recruited_in_dept,@H_Skills,@H_Behaviour,@H_Time_management)
Commit Transaction
Exec Proc_HRQ_Insert '5', 'YES', '2','YES','GOOD','GOOD','OKAY'

Select*from HRQ
